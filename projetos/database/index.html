1 Consulta de Clientes e Contratos Recuperar o nome e telefone dos clientes, número de contrato, modelo e placa do veículo. Ordenado pela data de saída mais recente
SQL 

SELECT 
    c.nome AS NomeCliente,
    t.numero AS Telefone,
    ct.numero AS NumeroContrato,
    v.modelo AS ModeloVeiculo,
    v.placa AS PlacaVeiculo,
    ct.dataSaida AS DataSaida
FROM 
    cliente c
LEFT JOIN 
    telefone t ON c.codigo = t.codigoCliente
JOIN 
    contrato ct ON c.codigo = ct.codigoCliente
JOIN 
    veiculo v ON ct.idVeiculo = v.id
ORDER BY 
    ct.dataSaida DESC;
1 Consulta de Clientes e Contratos Recuperar o nome e telefone dos clientes, número de contrato, modelo e placa do veículo. Ordenado pela data de saída mais recente
SQL 

SELECT 
    f.codigo AS CodigoFilial,
    ci.nome AS Cidade,
    ci.anoFundacao AS AnoFundacao,
    COUNT(v.id) AS TotalVeiculos
FROM 
    filial f
JOIN 
    cidade ci ON f.idCidade = ci.codigo
LEFT JOIN 
    veiculo v ON f.codigo = v.codFilial
GROUP BY 
    f.codigo, ci.nome, ci.anoFundacao
ORDER BY 
    f.codigo;

3. Contratos de Filomena com Veículo MAB0002

SQL 

SELECT 
    c.nome AS NomeCliente,
    ct.numero AS NumeroContrato,
    v.placa AS PlacaVeiculo,
    v.modelo AS ModeloVeiculo,
    ct.dataSaida AS DataSaida,
    ct.dataDevolucao AS DataDevolucao
FROM 
    cliente c
JOIN 
    contrato ct ON c.codigo = ct.codigoCliente
JOIN 
    veiculo v ON ct.idVeiculo = v.id
WHERE 
    c.nome = 'Filomena' AND v.placa = 'MAB0002';

4. Veículos de 4 Portas em Cidades Pequenas

SQL 

SELECT 
    v.placa AS Placa,
    v.modelo AS Modelo,
    ci.nome AS Cidade,
    f.rua AS RuaFilial
FROM 
    veiculo v
JOIN 
    filial f ON v.codFilial = f.codigo
JOIN 
    cidade ci ON f.idCidade = ci.codigo
WHERE 
    v.numPortas = 4 AND ci.populacao < 100000;

5. Veículos Caros em Cidades Grandes

SQL 

SELECT 
    v.modelo AS Modelo,
    COUNT(*) AS Quantidade
FROM 
    veiculo v
JOIN 
    filial f ON v.codFilial = f.codigo
JOIN 
    cidade ci ON f.idCidade = ci.codigo
WHERE 
    v.valorDiaria > 100 AND ci.populacao > 500000
GROUP BY 
    v.modelo;

6. Média de Valores por Veículo

SQL 

SELECT 
    v.placa AS Placa,
    v.modelo AS Modelo,
    AVG(ct.valorPago) AS MediaValorRecebido
FROM 
    veiculo v
JOIN 
    contrato ct ON v.id = ct.idVeiculo
GROUP BY 
    v.placa, v.modelo
ORDER BY 
    MediaValorRecebido DESC;

7. Veículos com Maiores Multas

SQL 

SELECT 
    v.placa AS Placa,
    v.modelo AS Modelo,
    v.valorDiaria AS ValorDiaria,
    ct.multa AS MaiorMulta
FROM 
    veiculo v
JOIN 
    contrato ct ON v.id = ct.idVeiculo
WHERE 
    ct.multa = (SELECT MAX(multa) FROM contrato)
ORDER BY 
    ct.multa DESC;

8. Veículos Nunca Locados

SQL 

SELECT 
    v.placa AS Placa,
    v.modelo AS Modelo
FROM 
    veiculo v
LEFT JOIN 
    contrato ct ON v.id = ct.idVeiculo
WHERE 
    ct.idVeiculo IS NULL;
